📋 COMMANDS:
═══════════════════════════════════════════════════════

# Start
docker-compose up -d

# Stop
docker-compose down

# Rebuild
docker-compose up -d --build

# Logs
docker-compose logs -f

# Restart
docker-compose restart

# Remove everything
docker-compose down -v
